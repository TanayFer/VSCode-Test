CREATE TABLE CUSTOMER (

);



CREATE TABLE BRANCH (

&nbsp;   BranchID INT PRIMARY KEY,

&nbsp;   BranchName VARCHAR(100) NOT NULL,

&nbsp;   City VARCHAR(50) NOT NULL

);



CREATE TABLE ACCOUNT (

&nbsp;   AccountNo INT PRIMARY KEY,

&nbsp;   AccountType VARCHAR(20),

&nbsp;   Balance DECIMAL(15, 2) CHECK (Balance >= 0),

&nbsp;   Status VARCHAR(20) DEFAULT 'Active',

&nbsp;   CustomerID INT,

&nbsp;   BranchID INT,

&nbsp;   FOREIGN KEY (CustomerID) REFERENCES CUSTOMER(CustomerID),

&nbsp;   FOREIGN KEY (BranchID) REFERENCES BRANCH(BranchID)

);



CREATE TABLE TRANSACTION (

&nbsp;   TransID INT PRIMARY KEY,

&nbsp;   TransDate DATE NOT NULL,

&nbsp;   Amount DECIMAL(15, 2) CHECK (Amount > 0),

&nbsp;   AccountNo INT,

&nbsp;   FOREIGN KEY (AccountNo) REFERENCES ACCOUNT(AccountNo)

);

CREATE TABLE DEPARTMENT (

CREATE TABLE USER (

&nbsp;   UserID INT PRIMARY KEY,

&nbsp;   UserName VARCHAR(100) NOT NULL,

&nbsp;   Email VARCHAR(150) UNIQUE,

&nbsp;   RegistrationDate DATETIME DEFAULT CURRENT\_TIMESTAMP

);



CREATE TABLE PRODUCT (

&nbsp;   ProductID INT PRIMARY KEY,

&nbsp;   ProductName VARCHAR(255) NOT NULL,

&nbsp;   Price DECIMAL(10, 2) NOT NULL CHECK (Price >= 0),

&nbsp;   Stock INT NOT NULL CHECK (Stock >= 0)

);



CREATE TABLE ORDER (

&nbsp;   OrderID INT PRIMARY KEY,

&nbsp;   OrderDate DATE NOT NULL,

&nbsp;   UserID INT,

&nbsp;   Status VARCHAR(20) DEFAULT 'Pending',

&nbsp;   FOREIGN KEY (UserID) REFERENCES USER(UserID)

);



CREATE TABLE ORDER\_ITEM (

&nbsp;   OrderID INT,

&nbsp;   ProductID INT,

&nbsp;   Quantity INT CHECK (Quantity > 0),

&nbsp;   PRIMARY KEY (OrderID, ProductID),

&nbsp;   FOREIGN KEY (OrderID) REFERENCES ORDER(OrderID),

&nbsp;   FOREIGN KEY (ProductID) REFERENCES PRODUCT(ProductID)

);
